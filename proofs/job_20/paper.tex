\section{Conjecture Statement}

\begin{conjecture}
\begin{lstlisting}
Can Single-Shuffle SGD be Better than Reshuffling SGD and GD?

Let n >= 2, K >= 1, and d >= 1. Let S_n denote the set of all permutations of {1,...,n}. For real symmetric d x d matrices X, Y, write X ⪯ Y for the Loewner order (i.e., Y - X is positive semidefinite), and let ||·||_2 be the spectral (operator) norm.

Given symmetric matrices A_1, ..., A_n, define for each σ ∈ S_n the without-replacement product P_σ := ∏_{i=1}^{n} A_{σ(i)} = A_{σ(n)} ⋯ A_{σ(1)}.

Define:
  W_SS := (1/n!) Σ_{σ ∈ S_n} (P_σ)^K       (single-shuffle)
  W_RS := ((1/n!) Σ_{σ ∈ S_n} P_σ)^K        (random reshuffling)
  W_GD := ((1/n) Σ_{i=1}^{n} A_i)^{nK}      (gradient descent)

CONJECTURE: For every n >= 2 and K >= 1 there exists a constant η_{n,K} ∈ (0,1] such that, whenever (1 - η_{n,K})I ⪯ A_i ⪯ I for all i ∈ {1,...,n}, one has:

  ||W_SS||_2 ≤ ||W_RS||_2 ≤ ||W_GD||_2
\end{lstlisting}
\end{conjecture}

# Helpers
# ---------------------------------------------------------------
def random_sym_in_interval(d, eta, rng):
    """Symmetric d x d with all eigenvalues in [1-eta, 1]."""
    # random orthogonal via QR
    G = rng.standard_normal((d, d))
    Q, R = np.linalg.qr(G)
    # fix sign to make Q uniform on O(d)
    Q = Q * np.sign(np.diag(R))
    evals = 1.0 - eta * rng.random(d)   # in [1-eta, 1]
    return (Q * evals) @ Q.T

def spectral_norm(M):
    # M is complex-or-real; we use 2-norm (largest singular value)
    return np.linalg.norm(M, ord=2)

def compute_W(A_list, K):
    """Return ||W_SS||, ||W_RS||, ||W_GD||."""
    n = len(A_list)
    d = A_list[0].shape[0]
    # all permutations
    perms = list(itertools.permutations(range(n)))
    # P_sigma = A_{sigma(n)} ... A_{sigma(1)}; i.e. multiply left-to-right
    # using sigma(1), then sigma(2), ... so result = A_{sigma(n)} @ ... @ A_{sigma(1)}.
    # Equivalent: start with I, then for idx in sigma left-to-right do  M = A[idx] @ M
    P_sum = np.zeros((d, d))
    PK_sum = np.zeros((d, d))
    for sigma in perms:
        M = np.eye(d)
        for idx in sigma:
            M = A_list[idx] @ M
        P_sum += M
        PK_sum += np.linalg.matrix_power(M, K)
    invfact = 1.0 / math.factorial(n)
    W_SS = invfact * PK_sum
    W_RS = np.linalg.matrix_power(invfact * P_sum, K)
    Abar = sum(A_list) / n
    W_GD = np.linalg.matrix_power(Abar, n * K)
    return spectral_norm(W_SS), spectral_norm(W_RS), spectral_norm(W_GD)
